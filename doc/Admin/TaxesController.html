def create
  @tax = Tax.new(tax_params)
  respond_to do |format|
    if @tax.save
      format.html { redirect_to admin_taxes_path, notice: 'Imposto registrado com sucesso.' }
    else
      format.html { redirect_to admin_taxes_path, notice: 'Não registrado. Tente novamente.' }
    end
  end
end
