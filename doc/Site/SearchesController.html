def create
  @search = Search.new(search_params)
  @search.user_id = current_user.id if user_signed_in?

  respond_to do |format|
    if @search.save
      format.html { redirect_to searches_path(termo: "#{@search.term}") }
    else
      format.html { redirect_to searches_path(termo: "#{@search.term}") }
    end
  end
end
